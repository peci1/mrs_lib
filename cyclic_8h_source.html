108  static flt wrap(const flt val)
109  {
110  // these few ifs should cover most cases, improving speed and precision
111  if (val >= minimum)
112  {
113  if (val < supremum) // value is actually in range and doesn't need to be wrapped
114  return val;
115  else if (val < supremum + range)
116  return val - range; // to avoid unnecessary costly fmod operation for this case (assumed to be significantly more common than the general case)
117  } else
118  {
119  if (val >= minimum - range)
120  return val + range; // to avoid unnecessary costly fmod operation for this case (assumed to be significantly more common than the general case)
121  }
122 
123  // general case
124  const flt rem = std::fmod(val - minimum, range);
125  const flt wrapped = rem + minimum + std::signbit(rem) * range;
126  return wrapped;
