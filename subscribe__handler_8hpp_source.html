108  /* waitForNew() method //{ */
109  virtual typename MessageType::ConstPtr waitForNew(const ros::WallDuration& timeout)
110  {
111  // convert the ros type to chrono type
112  const std::chrono::duration<float> chrono_timeout(timeout.toSec());
113  // lock the mutex guarding the m_new_data flag
114  std::unique_lock lock(m_new_data_mtx);
115  // if new data is available, return immediately, otherwise attempt to wait for new data using the respective condition variable
116  if (m_new_data)
117  return getMsg();
118  else if (m_new_data_cv.wait_for(lock, chrono_timeout) == std::cv_status::no_timeout && m_new_data)
119  return getMsg();
120  else
121  return nullptr;
122  };

328 
329  protected:
330  virtual void data_callback(const typename MessageType::ConstPtr& msg) override
331  {
332  // the stop has to come before the mutex lock to enable the timer callbacks currently
333  // in the queue to execute and prevent deadlock (.stop() waits for all currently
334  // queued callbacks to finish before returning...)
335  if (this->m_timeout_check_timer)
336  this->m_timeout_check_timer->stop();
337  std::scoped_lock lck(m_mtx, this->m_new_data_mtx);
338  this->process_new_message(msg);
339  if (this->m_message_callback)
340  this->m_message_callback(*(this->m_owner));
341  if (this->m_timeout_check_timer)
342  this->m_timeout_check_timer->start();
343  }
344

347  };
