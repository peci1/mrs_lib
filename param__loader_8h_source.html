446  /* load helper function for generic types //{ */
447  // This function tries to load a parameter with name 'name' and a default value.
448  // You can use the flag 'optional' to not throw a ROS_ERROR when the parameter
449  // cannot be loaded and the flag 'printValues' to set whether the loaded
450  // value and name of the parameter should be printed to cout.
451  // If 'optional' is set to false and the parameter could not be loaded,
452  // the flag 'load_successful' is set to false and a ROS_ERROR message
453  // is printer.
454  // If 'unique' flag is set to false then the parameter is not checked
455  // for being loaded twice.
456  // Returns a tuple, containing either the loaded or the default value and a bool,
457  // indicating if the value was loaded (true) or the default value was used (false).
458  template <typename T>
459  std::pair<T, bool> load(const std::string& name, const T& default_value, optional_t optional = OPTIONAL, unique_t unique = UNIQUE)
460  {
461  const std::string name_prefixed = m_prefix + name;
462  T loaded = default_value;
463  if (unique && check_duplicit_loading(name_prefixed))
464  return {loaded, false};
465 
466  bool cur_load_successful = true;
467  // try to load the parameter
468  const bool success = m_nh.getParam(name_prefixed, loaded);
469  if (!success)
470  {
471  // if it was not loaded, set the default value
472  loaded = default_value;
473  if (!optional)
474  {
475  // if the parameter was compulsory, alert the user and set the flag
476  printError(std::string("Could not load non-optional parameter ") + resolved(name_prefixed));
477  cur_load_successful = false;
478  }
479  }
480 
481  if (cur_load_successful)
482  {
483  // everything is fine and just print the name_prefixed and value if required
484  if (m_print_values)
485  printValue(name_prefixed, loaded);
486  // mark the param name_prefixed as successfully loaded
487  loaded_params.insert(name_prefixed);
488  } else
489  {
490  m_load_successful = false;
491  }
492  // finally, return the resulting value
493  return {loaded, success};
494  }
